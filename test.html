# ================================================================
# CELL 3 — Define crime tiers and aggregate to ward level
# ================================================================
type1_cats = [
    'VIOLENCE AGAINST THE PERSON', 'ROBBERY',
    'SEXUAL OFFENCES', 'POSSESSION OF WEAPONS'
]
type2_cats = [
    'THEFT', 'BURGLARY',
    'VEHICLE OFFENCES', 'ARSON AND CRIMINAL DAMAGE'
]
type3_cats = [
    'DRUG OFFENCES', 'PUBLIC ORDER OFFENCES',
    'MISCELLANEOUS CRIMES AGAINST SOCIETY', 'FRAUD AND FORGERY'
]
# NFIB FRAUD dropped — online crime, not spatially meaningful at ward level

# Sum monthly columns to get annual count per row
month_cols = [c for c in crime_raw.columns if str(c).startswith("2")]
crime_raw["annual_count"] = crime_raw[month_cols].sum(axis=1)

def aggregate_crime_type(df, major_types, count_col):
    """Filter to crime types, sum annual counts per ward."""
    filtered = df[df["MajorText"].isin(major_types)].copy()
    grouped = (
        filtered
        .groupby(["WardCode", "WardName", "LookUp_BoroughName"])["annual_count"]
        .sum()
        .reset_index()
        .rename(columns={
            "annual_count": count_col,
            "LookUp_BoroughName": "Borough"
        })
    )
    return grouped

crime_t1 = aggregate_crime_type(crime_raw, type1_cats, "type1_count")
crime_t2 = aggregate_crime_type(crime_raw, type2_cats, "type2_count")
crime_t3 = aggregate_crime_type(crime_raw, type3_cats, "type3_count")

# Merge all three together
ward_crime = (
    crime_t1
    .merge(crime_t2[["WardCode", "type2_count"]], on="WardCode", how="left")
    .merge(crime_t3[["WardCode", "type3_count"]], on="WardCode", how="left")
    .fillna(0)
)

print(f"Wards in crime data: {len(ward_crime)}")

# CELL 5 — Load ward boundary polygons from GitHub shapefile
# ================================================================
shp_base = (
    "https://raw.githubusercontent.com/EmilyDeeb/0006_Assessment"
    "/main/data/London-wards-2018_ESRI/"
)

tmp_dir = tempfile.mkdtemp()
for ext in ["shp", "dbf", "shx", "prj", "cpg"]:
    url = shp_base + f"London_Ward.{ext}"
    r = requests.get(url)
    with open(os.path.join(tmp_dir, f"London_Ward.{ext}"), "wb") as f:
        f.write(r.content)

wards_gdf = gpd.read_file(os.path.join(tmp_dir, "London_Ward.shp"))

wards_gdf = wards_gdf.rename(columns={
    "GSS_CODE": "WardCode",
    "NAME":     "WardName_geo",
    "DISTRICT": "Borough_geo"
})

# Area from HECTARES column (already in file)
wards_gdf["area_km2"] = wards_gdf["HECTARES"] / 100

ward_df = ward_crime.merge(
    wards_gdf[["WardCode", "area_km2", "geometry"]],
    on="WardCode", how="left"
)

# Crime densities — crimes per km² (target variables)
ward_df["type1_density"] = ward_df["type1_count"] / ward_df["area_km2"]
ward_df["type2_density"] = ward_df["type2_count"] / ward_df["area_km2"]
ward_df["type3_density"] = ward_df["type3_count"] / ward_df["area_km2"]

# Population density — may use as control variable
